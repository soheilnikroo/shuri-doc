Package Definition (index.ts)

How to define your package metadata and export commands

ğŸ“ packages/my-package/index.ts

import type { IPackageDefinition } from '@/types';

import setupCommand from '/commands/setup';
import buildCommand from '/commands/build';
import deployCommand from '/commands/deploy';
import { migrations } from '/commands';

const packageDefinition: IPackageDefinition = {
  metadata: {
    name: 'my-package',
    version: '1.2.0',
    description: 'A powerful development toolkit for modern applications',
  },
  commands: [
    setupCommand,
    buildCommand,
    deployCommand,
  ],
  migrations,
};

export default packageDefinition;
